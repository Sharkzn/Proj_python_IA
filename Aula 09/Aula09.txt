import nltk
from nltk.corpus import stopwords
from nltk.tokenize import word_tokenize
from textblob import TextBlob

"""Baixar os componentes NLTK necessários"""

nltk.download('punkt')
nltk.download('stopwords')

"""Função de pré processamento de texto"""

def preprocess(texto):
  #Tokenização
  tokens = word_tokenize(texto.lower())

  #Remover os espaços das frases
  espacos = set(stopwords.words('english'))

  #Filtrando os tokens
  tokens_filtrados = [palavra for palavra in tokens if palavra.isalnum() and palavra not in espacos]

  #Retorno da função
  return ' '.join(tokens_filtrados)

"""Função analise de sentimentos"""

def sentimentos(texto):

  #Limpando o texto digitado
  texto_limpo = preprocess(texto)

  #Abalise de sentimentos do texto
  blob = TextBlob(texto_limpo)

  #Aplaicando a polaridade do sentimento
  sentimento = blob.sentiment.polarity

  #Retornando a pontuação
  return sentimento

"""Listas de frases com sentimento"""

frases = [
    "You make my heart smile every time I see you.",
    "Our love story is my favorite adventure.",
    "Every moment with you feels like a beautiful dream.",
    "You are the reason I believe in love at first sight.",
    "Your laughter is my favorite melody.",
    "With you, I've found a love that knows no bounds.",
    "You complete me in ways I never thought possible.",
    "Your presence in my life is a precious gift.",
    "Every day spent with you is a new page in our love story.",
    "I love you more than words can express."
]

"""Exibindo a frase e sua pontuação"""

for comentario in frases:
  nivel = sentimentos(comentario)
  print(f"Frase: {comentario}\nPontuação: {round(nivel,2)}\n")
